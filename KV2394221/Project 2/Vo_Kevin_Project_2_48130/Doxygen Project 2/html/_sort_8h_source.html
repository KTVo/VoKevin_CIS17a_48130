110  T arrStart = ptrArray[start], holder;
111  int com = (start - 1), mid = (end + 1);
112 
113  do{
114  do{ mid--; }while (arrStart > ptrArray[mid]);
115  do{ com++; }while (arrStart < ptrArray[com]);
116  if(com < mid){
117  holder = ptrArray[com];

120  }
121  }while(com < mid);
